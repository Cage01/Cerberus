11UWorld* UCerberusAttributeSet::GetWorld() const
12{
13 /*
14 *
15 * A UObject does not know inherently
16 * what world it belongs to. The purpose
17 * of the GetWorld function is for that
18 * Object to be able to let others know
19 * which World it is in. Actors do this
20 * via their Level reference and
21 * Components do this either via their
22 * Actor owner or the World theyâ€™ve been
23 * registered with. Each other Object
24 * type has to find its own path back to
25 * World, often through a reference to an
26 * Actor.
27 *
28 * If you want to be able to call
29 * GetWorld on your class, you need to
30 * override the function and provide a
31 * valid path to a World.
32 *
33 */
34
35 const UObject* Outer = GetOuter();
36 check(Outer);
37
38 return Outer->GetWorld();
39}
