50void UCerberusPawnExtensionComponent::InitializeAbilitySystem(UCerberusAbilitySystemComponent* InASC, AActor* InOwnerActor)
51{
52 check(InASC);
53 check(InOwnerActor);
54
55 // If the ability system hasn't changed just return
56 if (AbilitySystemComponent == InASC)
57 return;
58
59 // If it has changed, clean up the old ability system
60 if (AbilitySystemComponent)
61 UninitializeAbilitySystem();
62
63 APawn* Pawn = GetPawnChecked<APawn>();
64 AActor* ExistingAvatar = InASC->GetAvatarActor();
65
66 UE_LOG(LogCerberus, Verbose, TEXT("Setting up ASC [%s] on pawn [%s] owner [%s], existing [%s] "), *GetNameSafe(InASC), *GetNameSafe(Pawn), *GetNameSafe(InOwnerActor), *GetNameSafe(ExistingAvatar));
67
68 if ((ExistingAvatar != nullptr) && ExistingAvatar != Pawn)
69 {
70 UE_LOG(LogCerberus, Log, TEXT("Existing avatar (authority=%d)"), ExistingAvatar->HasAuthority() ? 1 : 0);
71
72 // There is already a pawn acting as the ASC's avatar, so we need to kick it out
73 // This can happen on clients if they're lagged: their new pawn is spawned + possessed before the dead one is removed
74 ensure(!ExistingAvatar->HasAuthority());
75
76 if (UCerberusPawnExtensionComponent* OtherExtensionComponent = FindPawnExtentionComponent(ExistingAvatar))
77 {
78 OtherExtensionComponent->UninitializeAbilitySystem();
79 }
80 }
81
82 AbilitySystemComponent = InASC;
83 AbilitySystemComponent->InitAbilityActorInfo(InOwnerActor, Pawn);
84
85 // if(ensure(PawnData))
86 // {
87 // // InASC->SetTagRelationshipMapping(PawnData->TagRelationshipMapping);
88 // }
89
90 OnAbilitySystemInitialized.Broadcast();
91}

93void UCerberusPawnExtensionComponent::UninitializeAbilitySystem()
94{
95 if (!AbilitySystemComponent)
96 {
97 return;
98 }
99
100 if (AbilitySystemComponent->GetAvatarActor() == GetOwner())
101 {
102 AbilitySystemComponent->CancelAbilities(nullptr, nullptr);
103 //AbilitySystemComponent->ClearAbilityInput();
104 AbilitySystemComponent->RemoveAllGameplayCues();
105
106 if (AbilitySystemComponent->GetOwnerActor() != nullptr)
107 {
108 AbilitySystemComponent->SetAvatarActor(nullptr);
109 }
110 else
111 {
112 AbilitySystemComponent->ClearActorInfo();
113 }
114
115 OnAbilitySystemUninitialized.Broadcast();
116 }
117
118 AbilitySystemComponent = nullptr;
119}

144bool UCerberusPawnExtensionComponent::CheckPawnReadyToInitialize()
145{
146 if (bPawnReadyToInitialize)
147 {
148 return true;
149 }
150
151 //Check Pawn data
152 // if (!PawnData)
153 // {
154 // return false;
155 // }
156
157 APawn* Pawn = GetPawnChecked<APawn>();
158
159 const bool bHasAuthority = Pawn->HasAuthority();
160 const bool bIsLocallyControlled = Pawn->IsLocallyControlled();
161
162 if (bHasAuthority || bIsLocallyControlled)
163 {
164 // Check for being possessed by a controller
165 if (!GetController<AController>())
166 {
167 return false;
168 }
169 }
170
171 // Allow pawn components to have requirements
172 TArray<UActorComponent*> InteractableComponents = Pawn->GetComponentsByInterface(UCerberusReadyInterface::StaticClass());
173 for (UActorComponent* Component : InteractableComponents)
174 {
175 const ICerberusReadyInterface* Ready = CastChecked<ICerberusReadyInterface>(Component);
176 if (!Ready->IsPawnComponentReadyToInitialize())
177 {
178 return false;
179 }
180 }
181
182 //Pawn is ready to initialize
183 bPawnReadyToInitialize = true;
184 OnPawnReadyToInitialize.Broadcast();
185 BP_OnPawnReadyToInitialize.Broadcast();
186
187 return true;
188}
