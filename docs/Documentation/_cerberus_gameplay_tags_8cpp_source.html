52FGameplayTag FCerberusGameplayTags::FindTagByString(FString TagString, bool bMatchPartialString)
53{
54 const UGameplayTagsManager& Manager = UGameplayTagsManager::Get();
55 FGameplayTag Tag = Manager.RequestGameplayTag(FName(*TagString), false);
56
57 if (!Tag.IsValid() && bMatchPartialString)
58 {
59 FGameplayTagContainer AllTags;
60 Manager.RequestAllGameplayTags(AllTags, true);
61
62 for (const FGameplayTag TestTag : AllTags)
63 {
64 if (TestTag.ToString().Contains(TagString))
65 {
66 UE_LOG(LogCerberus, Display, TEXT("Could not find exact match for tag [%s] but found partial match on tag [%s]."), *TagString, *TestTag.ToString());
67 Tag = TestTag;
68 break;
69 }
70 }
71 }
72
73 return Tag;
74}
